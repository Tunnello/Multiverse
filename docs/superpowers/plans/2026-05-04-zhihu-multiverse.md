# Zhihu Multiverse (观点对撞机) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a statically exported Next.js demo that loads three fixture graph JSON files from `public/data`, renders an AntV G6 force-directed graph with camp colors and clash edges, a topic picker (问题 1/2/3), a year slider for time drift (hide nodes with `publishedAt` year > selected year), and a right drawer for the selected node—plus a local FastAPI `pipeline` that can call Zhihu search (max 10) and (stub or real) LLM to regenerate those JSON files offline.

**Architecture:** Monorepo-style folders `web/` (Next `output: 'export'`, no API routes in prod) and `pipeline/` (FastAPI, dev-only). Shared truth for graph shape is `web/lib/graphSchema.ts` (Zod); `pipeline` mirrors the same fields in Pydantic for response validation. Demo JSON is generated by a checked-in Node script so CI and Vercel never depend on secrets.

**Tech Stack:** Next.js 15 (App Router), React 19, TypeScript, Zod, Vitest, @antv/g6, Tailwind CSS (optional but default from create-next-app), Python 3.11+, FastAPI, httpx, Pydantic v2.

export const CampEnum = z.enum([
  "academic",
  "radical",
  "experiential",
  "stakeholder",
]);

export const EdgeKindEnum = z.enum(["agree", "clash", "complement"]);

export const AuthorSchema = z.object({
  name: z.string(),
  badgeText: z.string().optional(),
  avatarUrl: z.string().url().optional().or(z.literal("")),
});

export const PredictionScoreSchema = z.object({
  predictionDeviation: z.union([z.number(), z.string()]),
  driftLabel: z.string().optional(),
});

export const GraphNodeSchema = z.object({
  id: z.string(),
  label: z.string(),
  camp: CampEnum,
  summary: z.string(),
  quote: z.string(),
  author: AuthorSchema,
  voteUpCount: z.number(),
  publishedAt: z.string(),
  predictionScore: PredictionScoreSchema,
  contentType: z.string().optional(),
});

export const GraphEdgeSchema = z.object({
  id: z.string(),
  source: z.string(),
  target: z.string(),
  kind: EdgeKindEnum,
  label: z.string(),
});

export const GraphDocumentSchema = z.object({
  schemaVersion: z.string(),
  topic: z.object({
    title: z.string(),
    slug: z.string(),
    sourceNote: z.string().optional(),
  }),
  timeRange: z.object({
    minYear: z.number().int(),
    maxYear: z.number().int(),
  }),
  nodes: z.array(GraphNodeSchema),
  edges: z.array(GraphEdgeSchema),
});

export type GraphDocument = z.infer<typeof GraphDocumentSchema>;
export type GraphNode = z.infer<typeof GraphNodeSchema>;
export type GraphEdge = z.infer<typeof GraphEdgeSchema>;

- [ ] **Step 3: Implement `web/lib/filterByYear.ts`**

```ts
import type { GraphDocument, GraphEdge, GraphNode } from "@/lib/graphSchema";

function yearFromPublishedAt(publishedAt: string): number {
  const d = new Date(publishedAt);
  if (Number.isNaN(d.getTime())) return Number.NaN;
  return d.getUTCFullYear();
}

export function filterGraphByYear(doc: GraphDocument, selectedYear: number) {
  const visibleNodeIds = new Set<string>();
  for (const n of doc.nodes) {
    const y = yearFromPublishedAt(n.publishedAt);
    if (!Number.isFinite(y)) continue;
    if (y <= selectedYear) visibleNodeIds.add(n.id);
  }
  const edges: GraphEdge[] = [];
  for (const e of doc.edges) {
    if (visibleNodeIds.has(e.source) && visibleNodeIds.has(e.target)) {
      edges.push(e);
    }
  }
  const nodes: GraphNode[] = doc.nodes.filter((n) => visibleNodeIds.has(n.id));
  return { nodes, edges, visibleNodeIds };
}
```

```tsx
"use client";

import { useEffect, useRef } from "react";
import { Graph } from "@antv/g6";
import type { GraphEdge, GraphNode } from "@/lib/graphSchema";
import { CAMP_COLOR } from "@/lib/campColors";

type Props = {
  nodes: GraphNode[];
  edges: GraphEdge[];
  onSelect: (id: string | null) => void;
};

export function GraphCanvas({ nodes, edges, onSelect }: Props) {
  const ref = useRef<HTMLDivElement>(null);
  const graphRef = useRef<Graph | null>(null);

  useEffect(() => {
    if (!ref.current) return;
    const g6Nodes = nodes.map((n) => ({
      id: n.id,
      data: { ...n, color: CAMP_COLOR[n.camp] },
    }));
    const g6Edges = edges.map((e) => ({
      id: e.id,
      source: e.source,
      target: e.target,
      data: { ...e, lineDash: e.kind === "clash" ? [6, 4] : undefined },
    }));

    const graph = new Graph({
      container: ref.current,
      width: ref.current.clientWidth,
      height: ref.current.clientHeight,
      data: { nodes: g6Nodes, edges: g6Edges },
      node: {
        style: { fill: (d: { data?: { color?: string } }) => d.data?.color ?? "#888" },
      },
      edge: {
        style: {
          stroke: (d: { data?: { kind?: string } }) =>
            d.data?.kind === "clash" ? "#F87171" : "#94A3B8",
          lineDash: (d: { data?: { lineDash?: number[] } }) => d.data?.lineDash,
        },
      },
      layout: { type: "force", linkDistance: 120, clustering: true },
      behaviors: ["drag-canvas", "zoom-canvas", "drag-element"],
    });

    graph.render();
    graph.on("node:click", (ev: { target?: { id: string } }) => {
      const id = ev.target?.id;
      onSelect(id ?? null);
    });
    graphRef.current = graph;

    return () => {
      graph.destroy();
      graphRef.current = null;
    };
  }, [nodes, edges, onSelect]);

  return <div ref={ref} className="h-full w-full min-h-[480px]" />;
}
```
